172 typedef struct OptionDef {
173  const char *name;
174  int flags;
175 #define HAS_ARG 0x0001
176 #define OPT_BOOL 0x0002
177 #define OPT_EXPERT 0x0004
178 #define OPT_STRING 0x0008
179 #define OPT_VIDEO 0x0010
180 #define OPT_AUDIO 0x0020
181 #define OPT_INT 0x0080
182 #define OPT_FLOAT 0x0100
183 #define OPT_SUBTITLE 0x0200
184 #define OPT_INT64 0x0400
185 #define OPT_EXIT 0x0800
186 #define OPT_DATA 0x1000
187 #define OPT_PERFILE 0x2000 /* the option is per-file (currently ffmpeg-only).
188  implied by OPT_OFFSET or OPT_SPEC */
189 #define OPT_OFFSET 0x4000 /* option is specified as an offset in a passed optctx */
190 #define OPT_SPEC 0x8000 /* option is to be stored in an array of SpecifierOpt.
191  Implies OPT_OFFSET. Next element after the offset is
192  an int containing element count in the array. */
193 #define OPT_TIME 0x10000
194 #define OPT_DOUBLE 0x20000
195 #define OPT_INPUT 0x40000
196 #define OPT_OUTPUT 0x80000
197  union {
198  void *dst_ptr;
199  int (*func_arg)(void *, const char *, const char *);
200  size_t off;
201  } u;
202  const char *help;
203  const char *argname;
204 } OptionDef;
